16  //default port is 1500.
17  //connection to be closed by client.
18  //this server handles only 1 connection.
19  
20  import java.net.*;
21  
22  
23  class tcpServer {
24   
25   public static void main(String args[]) {
26  	
27  	int port;
28  	ServerSocket server_socket;
29  	BufferedReader input;
30  	try { 
31  	    port = Integer.parseInt(args[0]);
32  	}
33  	catch (Exception e) {
34  	    System.out.println("port = 1500 (default)");
35  	    port = 1500;
36  	}
37  	try {
38  	    
39  	    server_socket = new ServerSocket(port);
40  	    System.out.println("Server waiting for client on port " + 
41  			       server_socket.getLocalPort());
42  	    
43  	    // server infinite loop
44  	    while(true) {
45  		Socket socket = server_socket.accept();
46  		System.out.println("New connection accepted " +
47  				   socket.getInetAddress() +
48  				   ":" + socket.getPort());
49  		input = new BufferedReader(new InputStreamReader(socket.getInputStream())); 
50  		// print received data 
51  			try {
52  				while(true) {
53  				String message = input.readLine();
54  				if (message==null) break;
55  				System.out.println(message);
56  				}
57  			}
58  			catch (IOException e) {
59  				System.out.println(e);
60  			}
61  			// connection closed by client
62  			try {
63  				socket.close();
64  				System.out.println("Connection closed by client");
65  			}
66  			catch (IOException e) {
67  				System.out.println(e);
68  			}
69  	    } 
70  	}
71  	
72  	catch (IOException e) {
73  	    System.out.println(e);
74  	}
75   }
76  }
