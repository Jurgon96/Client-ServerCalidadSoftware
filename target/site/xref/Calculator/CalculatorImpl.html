6   package Calculator;
7   public class CalculatorImpl
8       extends 
9   	java.rmi.server.UnicastRemoteObject
10      implements Calculator {
11  
12      // Implementations must have an 
13      //explicit constructor
14      // in order to declare the 
15      //RemoteException exception
16      public CalculatorImpl()
17  	throws java.rmi.RemoteException {
18  	super();
19      }
20  
21      public long add(long a, long b)
22  	throws java.rmi.RemoteException {
23  	System.out.println("Sumando " + a + " " + b);
24  	return a + b;
25      }
26  
27      public long sub(long a, long b)
28  	throws java.rmi.RemoteException {
29  	System.out.println("Restando " + a + " " + b);
30  	return a - b;
31      }
32  
33      public long mul(long a, long b)
34  	throws java.rmi.RemoteException {
35  	System.out.println("Multiplicando " + a + " " + b);
36  	return a * b;
37      }
38  
39      public long div(long a, long b)
40  	throws java.rmi.RemoteException {
41  	System.out.println("Dividiendo " + a + " " + b);
42  
43  	return a / b;
44      }
45  	
46  	public long negativo(long a)
47  	throws java.rmi.RemoteException {
48  	System.out.println(")Invirtiendo numero " + a);	
49  		return a*(-1);
50  	}
51  	
52    public void hello() {
53        System.out.println("NANA");
54    }
55  }
